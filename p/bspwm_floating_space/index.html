bspwm添加全部浮动的空间

例如: 希望添加到第10号的空间全部都是浮动的

把下面的脚本添加进bspwmrc中

#! /bin/bash

FLOATING_DESKTOP_ID=$(bspc query -D -d '^10')

bspc subscribe node | while read -a msg ; do
    desk_id=${msg[2]}
    wid=${msg[3]}
    [ "$FLOATING_DESKTOP_ID" = "$desk_id" ] && bspc node "$wid" -t floating
done
